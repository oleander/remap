def call(state, &block)
  unless block
    return call(state, &:itself)
  end

  state.bind(quantifier: "*") do |enumerable, inner_state, &error|
    requirement[enumerable] do
      return error["Expected an enumeration"]
    end

    inner_state.map(&block)
  end
end
