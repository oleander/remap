def self.call(&block)

  end

  new.tap { _1.execute(&block) }

def get(key, &error)
  unless error
    return get(key) { raise Error, _1 }
  end

  table.fetch(key) do
    unless default == Undefined
      return default
    end

    error["Enum key [#{key}] not found among [#{table.keys.inspect}]"]
  end
end
