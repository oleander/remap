Overview

Sets the value to a given path

Examples:

Maps “A” to { a: { b: { c: “A” } } }

state = Remap::State.call("A")
result = Remap::Path::Output.new([:a, :b, :c]).call(state)

result.fetch(:value) # => { a: { b: { c: "A" } } }

def call(state)
  mapped = selectors.map do |selector|
    selector.call(state)
  end

  state.fmap do |value|
    mapped.hide(value)
  end
end
